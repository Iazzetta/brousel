brousel.js

A minimalist slider / carousel library, with high performance and extremely light (1.8kb).

open source


new Brousel( document.querySelector('#element') , {
    // enable/disable arrows navigation
    arrows: true,
    // enable/disable dots navigation
    dots: true,
    // slides to show each time in desktop
    slidesToShow: 3,
    // autoplay slide
    autoplay: true,
    // slide change speed
    speed: 5000,
    // margin top/bottom
    margin_horizontal: 5,
    // margin right/left
    margin_vertical: 0,
    // slides to show each time in mobile (size <= responsiveSizeToChange)
    responsiveToShow: 1,
    // width size to change slidesToShow to responsiveToShow
    responsiveSizeToChange: 600,
    // content of previous slide button
    prev_content: '<',
    // content of next slide button
    next_content '>',
    // content of dots
    dot_content: '-'
})
